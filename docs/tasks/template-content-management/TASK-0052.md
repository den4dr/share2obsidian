### 2. SettingsViewModel（新規） 🔵

**信頼性**: 🔵 *interfaces.kt SettingsViewModel・dataflow.md フロー3 より*

```kotlin
@HiltViewModel
class SettingsViewModel @Inject constructor(
    private val noteSettingsRepository: NoteSettingsRepository,
) : ViewModel() {
    val uiState: StateFlow<SettingsUiState> = noteSettingsRepository.getSettings()
        .map { SettingsUiState(it.vault, it.folder) }
        .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), SettingsUiState())

    fun updateVault(vault: String) = viewModelScope.launch(Dispatchers.IO) {
        noteSettingsRepository.saveVault(vault)
    }
    fun updateFolder(folder: String) = viewModelScope.launch(Dispatchers.IO) {
        noteSettingsRepository.saveFolder(folder)
    }
}
```
